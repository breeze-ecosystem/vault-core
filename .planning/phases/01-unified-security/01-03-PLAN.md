<objective>
Deliver the video-event correlation and unified timeline vertical slice. By the end of this plan, a security operator can view a unified timeline that merges access events, door state changes, and video clips in chronological order, click any event to see the associated video, monitor a real-time event stream with thumbnails, and search events by multiple filters. Additionally, tailgating detection triggers async AI analysis after valid access events and generates alerts when multiple persons are detected entering on a single credential.

Purpose: This is the core UX flow for Phase 1 — "click any access event → see the video clip." The unified timeline transforms Oversight Hub from a video surveillance platform into an integrated physical security intelligence system. Video correlation is the data layer that Phase 2 (incident management, visitor tracking, ANPR) builds on.

Output: Working unified timeline dashboard with video playback, real-time event stream via Socket.IO, tailgating detection integrated with existing AI pipeline, and searchable event history from TimescaleDB hypertables.
</objective>

    **Part C — Correlation Processor (correlation.processor.ts):**

    Implement the async video correlation worker (D-13: never blocks access decisions):

    ```typescript
    @Processor('video-correlation')
    export class CorrelationProcessor {
      @Process('correlate-video')
      async correlateVideo(job: Job<CorrelationJob>): Promise<void> {
        const { doorId, siteId, eventType, timestamp } = job.data;

        // 1. Query CameraDoorMap for cameras mapped to this door (D-14), sorted by priority ASC
        const cameraMaps = await this.prisma.cameraDoorMap.findMany({
          where: { doorId },
          include: { camera: true },
          orderBy: { priority: 'asc' },
        });
        if (cameraMaps.length === 0) return; // No camera mapped → skip correlation

        // 2. For the top-priority camera, find the nearest video frame
        const camera = cameraMaps[0].camera;
        const eventTime = new Date(timestamp);

        // 3. Extract a snapshot/frame at eventTime (±2 seconds)
        // Use existing IngestionService or direct FFmpeg call to extract a frame
        // For Phase 1 baseline: use the camera's lastSnapshotUrl if it's within 5 seconds of eventTime
        // If not, trigger async snapshot capture via IngestionService.captureSnapshot(cameraId)
        const snapshotUrl = camera.lastSnapshotUrl; // Phase 1 MVP: use existing snapshot

        // 4. Generate a thumbnail (resize to 320x180) using the same frame
        // For Phase 1: use the same snapshotUrl as thumbnail (the dashboard can resize with CSS)

        // 5. Store correlation result in access_events metadata (update the row)
        await this.prisma.$queryRaw`
          UPDATE access_events
          SET metadata = COALESCE(metadata, '{}'::jsonb) || ${
            JSON.stringify({
              correlation: {
                cameraId: camera.id,
                cameraName: camera.name,
                snapshotUrl,
                thumbnailUrl: snapshotUrl,
                correlatedAt: new Date().toISOString(),
              }
            })
          }::jsonb
          WHERE door_id = ${doorId}::uuid
            AND time >= ${eventTime}::timestamptz - INTERVAL '1 second'
            AND time <= ${eventTime}::timestamptz + INTERVAL '1 second'
          LIMIT 1
        `;

        // 6. Emit correlation.ready event for real-time Socket.IO push (VEC-04)
        this.eventEmitter.emit('correlation.ready', {
          doorId, cameraId: camera.id, snapshotUrl, thumbnailUrl: snapshotUrl, timestamp,
        });
      }
    }
    ```

    **Part D — Tailgating Processor (tailgating.processor.ts) — AI-04:**

    Implement per RESEARCH.md "Tailgating Detection Trigger" pattern (lines 1483-1546):

    ```typescript
    @Processor('tailgating-detection')
    export class TailgatingProcessor {
      constructor(
        private prisma: PrismaService,
        private alertService: AlertService,
        private configService: ConfigService,
        private httpClient: HttpService, // @nestjs/axios or native fetch
      ) {}

      @Process('detect-tailgating')
      async detect(job: Job<TailgatingJob>): Promise<void> {
        const { doorId, siteId, eventTimestamp } = job.data;

        // 1. Get cameras mapped to the door (D-14: CameraDoorMap)
        const cameraMaps = await this.prisma.cameraDoorMap.findMany({
          where: { doorId },
          include: { camera: true },
          orderBy: { priority: 'asc' },
        });
        if (cameraMaps.length === 0) return;

        // 2. For the primary camera, get a snapshot at current time (t+3s after access event)
        const camera = cameraMaps[0].camera;
        const snapshotUrl = camera.lastSnapshotUrl;
        if (!snapshotUrl) return; // No recent snapshot available

        // 3. Send to Ollama vision model for person counting (D-20, D-21)
        const ollamaUrl = this.configService.get('OLLAMA_BASE_URL', 'http://localhost:11434');
        const model = this.configService.get('OLLAMA_MODEL', 'moondream');

        const response = await fetch(`${ollamaUrl}/api/generate`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            model,
            prompt: 'Count the number of people visible in this doorway image. Return ONLY a JSON object with a single key "count" and an integer value. Example: {"count": 1}',
            images: [snapshotUrl],
            stream: false,
          }),
        });

        const result = await response.json();
        // Parse the response — Ollama returns a text response, extract JSON from it
        let count = 1;
        try {
          const parsed = JSON.parse(result.response);
          count = typeof parsed.count === 'number' ? parsed.count : 1;
        } catch {
          // If parsing fails, try regex extraction: /\d+/ from the response text
          const match = (result.response || '').match(/\d+/);
          count = match ? parseInt(match[0]) : 1;
        }

        // 4. D-21: If >1 person detected, generate tailgating alert
        if (count > 1) {
          // Apply deduplication: check Redis for tailgating cooldown per door (120s)
          const cooldownKey = `tailgating:cooldown:${doorId}`;
          const existing = await this.cacheManager.get(cooldownKey);
          if (existing) return; // Already alerted recently

          await this.alertService.create({
            title: `Détection de tailgating — Porte ${doorId}`,
            description: `${count} personnes détectées lors d'un seul accès valide.`,
            severity: 'HIGH',
            cameraId: camera.id,
            snapshotUrl: camera.lastSnapshotUrl,
            metadata: {
              doorId,
              siteId,
              personCount: count,
              eventTimestamp,
              detectedAt: new Date().toISOString(),
              type: 'tailgating',
            },
          });

          // Set 120s cooldown
          await this.cacheManager.set(cooldownKey, '1', 120);
        }
      }
    }
    ```
