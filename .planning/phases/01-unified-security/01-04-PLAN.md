<objective>
Deliver the audit and compliance vertical slice. By the end of this plan, every mutation operation across all Phase 1 modules produces an immutable, hash-chained audit entry stored in the TimescaleDB audit_log hypertable. An auditor can verify cryptographic chain integrity via the `/audit/verify` endpoint, export filtered audit reports, and the new AUDITOR role is enforced across all modules with read-only access to audit data.

Purpose: Audit immutability is the trust foundation of the platform. Security operators and compliance officers must be able to prove that no event record has been tampered with. The hash chain provides cryptographic tamper evidence without requiring a separate immutable database. Fine-grained roles ensure that only authorized personnel can modify system state while auditors have comprehensive read access.

Output: Working audit interceptor with hash-chain integrity, `/audit/verify` endpoint for chain validation, audit export with filters, and AUDITOR role enforcement across all Phase 1 modules.
</objective>

    Create `apps/api/migrations/timescaledb/up/006_audit_hash_chain_trigger.sql` implementing RESEARCH.md Pattern 8 (lines 1062-1099) EXACTLY:

    ```sql
    -- Enable pgcrypto extension if not already enabled
    CREATE EXTENSION IF NOT EXISTS pgcrypto;

    CREATE OR REPLACE FUNCTION audit_hash_chain_trigger()
    RETURNS TRIGGER AS $$
    DECLARE
        prev_hash TEXT;
    BEGIN
        -- Find the most recent hash for this entity (per-entity chain, NOT global)
        SELECT hash INTO prev_hash
        FROM audit_log
        WHERE entity = NEW.entity
          AND entity_id = NEW.entity_id
        ORDER BY time DESC, hash DESC
        LIMIT 1;

        -- Set previous_hash for verification
        NEW.previous_hash := prev_hash;

        -- Compute SHA-256: hash(previous_hash || content)
        -- content must be the canonical representation of the mutation
        NEW.hash := encode(
            digest(
                COALESCE(prev_hash, 'genesis') || NEW.content,
                'sha256'
            ),
            'hex'
        );

        RETURN NEW;
    END;
    $$ LANGUAGE plpgsql;

    -- Apply trigger to audit_log table
    DROP TRIGGER IF EXISTS trg_audit_hash_chain ON audit_log;
    CREATE TRIGGER trg_audit_hash_chain
        BEFORE INSERT ON audit_log
        FOR EACH ROW
        EXECUTE FUNCTION audit_hash_chain_trigger();
    ```

    This trigger automatically computes previous_hash and hash for every INSERT on audit_log. The application layer (AuditInterceptor) writes entity+entity_id+action+content — the trigger handles the cryptographic chain.

    ```typescript
    @Injectable()
    export class AuditInterceptor implements NestInterceptor {
      constructor(
        @InjectQueue('audit-write') private auditQueue: Queue,
      ) {}

      intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
        const request = context.switchToHttp().getRequest();
        const handler = context.getHandler();

        // Check for @Audited() decorator metadata on the handler
        const auditMeta: AuditMetadata | undefined = Reflect.getMetadata('audit', handler);
        if (!auditMeta) return next.handle(); // Skip non-auditable endpoints

        const startTime = Date.now();

        return next.handle().pipe(
          tap((result) => {
            // D-18: Capture mutation and enqueue audit write (non-blocking)
            const auditEntry = {
              entity: auditMeta.entity,
              entityId: result?.id || request.params?.id || 'unknown',
              action: auditMeta.action,
              userId: request.user?.id || 'system',
              siteId: request.user?.siteId || request.body?.siteId || null,
              changes: null as any, // Computed below
              ipAddress: request.ip || request.socket?.remoteAddress,
              userAgent: request.headers?.['user-agent'],
              timestamp: new Date().toISOString(),
            };

            // Capture before/after for UPDATE operations
            if (auditMeta.action === 'UPDATE' && auditMeta.captureChanges) {
              // The 'before' state can be captured in the controller and attached to request
              const beforeState = (request as any).__auditBeforeState;
              auditEntry.changes = {
                before: beforeState || null,
                after: this.sanitizeForAudit(result),
              };
            } else if (auditMeta.action === 'CREATE') {
              auditEntry.changes = { after: this.sanitizeForAudit(result) };
            } else if (auditMeta.action === 'DELETE') {
              auditEntry.changes = { before: this.sanitizeForAudit(request.params) };
            }

            // D-13: Audit writes are async — never block the response
            this.auditQueue.add('write-audit', auditEntry, {
              attempts: 3,
              backoff: { type: 'exponential', delay: 1000 },
              removeOnComplete: true,
            });
          }),
          catchError((error) => {
            // Log failed mutation attempts too (for security monitoring)
            this.auditQueue.add('write-audit', {
              entity: auditMeta.entity,
              entityId: request.params?.id || 'unknown',
              action: `${auditMeta.action}_FAILED`,
              userId: request.user?.id || 'system',
              siteId: request.user?.siteId,
              changes: { error: error.message },
              ipAddress: request.ip,
              timestamp: new Date().toISOString(),
            });
            return throwError(() => error);
          }),
        );
      }

      private sanitizeForAudit(data: any): any {
        // Remove sensitive fields before writing to audit log
        if (!data) return data;
        const sanitized = { ...data };
        delete sanitized.password;
        delete sanitized.pinHash;
        delete sanitized.refreshToken;
        delete sanitized.token;
        return sanitized;
      }
    }

    // Decorator: marks a handler for audit capture
    export const AUDIT_METADATA_KEY = 'audit';
    export interface AuditMetadata {
      entity: string;    // e.g., 'credential', 'door', 'access_level'
      action: string;    // 'CREATE', 'UPDATE', 'DELETE'
      captureChanges?: boolean;
    }
    export const Audited = (meta: AuditMetadata) => SetMetadata(AUDIT_METADATA_KEY, meta);
    ```

      ```typescript
      async verifyChain(entity: string, entityId: string): Promise<ChainVerificationResult> {
        const entries = await this.prisma.$queryRaw<AuditEntry[]>`
          SELECT time, hash, previous_hash, content
          FROM audit_log
          WHERE entity = ${entity} AND entity_id = ${entityId}::uuid
          ORDER BY time ASC
        `;

        const tampered: number[] = [];
        const crypto = await import('crypto');

        for (let i = 0; i < entries.length; i++) {
          const expectedInput = (i === 0 ? 'genesis' : entries[i-1].hash) + entries[i].content;
          const expectedHash = crypto.createHash('sha256').update(expectedInput).digest('hex');

          if (entries[i].hash !== expectedHash) {
            tampered.push(i);
          }
          if (i > 0 && entries[i].previous_hash !== entries[i-1].hash) {
            if (!tampered.includes(i)) tampered.push(i);
          }
        }

        return {
          verified: tampered.length === 0,
          totalEntries: entries.length,
          tamperedIndices: tampered,
          genesisHash: entries[0]?.hash || null,
          latestHash: entries[entries.length - 1]?.hash || null,
        };
      }
      ```
