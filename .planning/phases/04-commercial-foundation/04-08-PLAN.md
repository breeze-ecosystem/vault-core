         const res = await fetchWithAuth(`${API_URL}/api/organizations/${orgId}/invites`);

       }

         const res = await fetchWithAuth(`${API_URL}/api/organizations/${orgId}/invites`, { method: "POST", body: JSON.stringify(data) });
