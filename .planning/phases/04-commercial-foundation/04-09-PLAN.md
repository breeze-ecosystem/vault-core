       export async function switchOrganization(orgId: string): Promise<AuthResult & { organization?: any }> {
         const token = await getAccessTokenAsync();
         const res = await fetch(`${API_URL}/api/auth/switch-org`, {
           method: "POST",
           headers: { "Content-Type": "application/json", ...(token ? { Authorization: `Bearer ${token}` } : {}) },
           credentials: "include",
           body: JSON.stringify({ organizationId: orgId }),
         });
         const data = await res.json();
         if (!res.ok) return { error: data.message || "Échec du changement d'organisation" };
         // Store tokens via SecureStore
         await saveTokens(data.accessToken, data.refreshToken);
         await saveUser(data.user);
         await saveOrganization(data.organization);
         return { accessToken: data.accessToken, user: data.user, organization: data.organization };
       }

|----------|-------------|
| Mobile app → API | JWT token stored in expo-secure-store; sent as Bearer in Authorization header |
| Auth context → SecureStore | Tokens and org data persisted in device keychain (expo-secure-store) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-38 | Elevation of Privilege | Mobile switch-org — stale token | mitigate | switchOrganization() re-issues tokens with new orgId. Old tokens invalidated server-side (refresh tokens revoked). Mobile stores new tokens in SecureStore. |
| T-04-39 | Information Disclosure | Mobile SecureStore — org data | accept | expo-secure-store uses device keychain (iOS) / EncryptedSharedPreferences (Android). Data is encrypted at rest. Accessible only by the app. |
</threat_model>
